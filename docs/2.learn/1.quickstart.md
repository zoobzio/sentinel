# Quickstart

## Installation

```bash
go get github.com/zoobzio/sentinel@latest
```

Requires Go 1.23 or later.

## Basic Usage

```go
package main

import (
    "fmt"
    "github.com/zoobzio/sentinel"
)

type User struct {
    ID    string `json:"id" db:"user_id"`
    Name  string `json:"name" validate:"required"`
    Email string `json:"email" validate:"email"`
}

func main() {
    // Extract metadata (cached after first call)
    metadata := sentinel.Inspect[User]()

    fmt.Printf("Type: %s\n", metadata.TypeName)
    fmt.Printf("Fields: %d\n", len(metadata.Fields))

    for _, field := range metadata.Fields {
        fmt.Printf("  %s: %s\n", field.Name, field.Type)
        for tag, value := range field.Tags {
            fmt.Printf("    @%s = %s\n", tag, value)
        }
    }
}
```

## Discovering Relationships

```go
type User struct {
    ID      string   `json:"id"`
    Profile *Profile `json:"profile"`
    Orders  []Order  `json:"orders"`
}

type Profile struct {
    Bio string `json:"bio"`
}

type Order struct {
    ID    string  `json:"id"`
    Total float64 `json:"total"`
}

// Scan recursively extracts User and all related types
metadata := sentinel.Scan[User]()

// Check relationships
for _, rel := range metadata.Relationships {
    fmt.Printf("%s -> %s (%s)\n", rel.From, rel.To, rel.Kind)
}
// Output:
// User -> Profile (reference)
// User -> Order (collection)

// All related types are now cached
types := sentinel.Browse()
// ["User", "Profile", "Order"]
```

## Registering Custom Tags

```go
// Register before first Inspect/Scan call
sentinel.Tag("custom")
sentinel.Tag("myapp")

type Model struct {
    Field string `custom:"value" myapp:"data"`
}

metadata := sentinel.Inspect[Model]()
// metadata.Fields[0].Tags["custom"] == "value"
// metadata.Fields[0].Tags["myapp"] == "data"
```
